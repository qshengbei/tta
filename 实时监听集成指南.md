### 修改 `pages/order-list/index.js`

```javascript
import { OrderListener } from '../../utils/orderListener';

Page({
  data: {
    orders: [],
    originalOrders: [],
    selectedStatus: "all",
    loading: true,
    // ... 其他数据
  },

  onLoad(options) {
    // 获取URL参数
    if (options && options.status) {
      this.setData({ selectedStatus: options.status });
    }

    // 初始化订单监听器
    this.orderListener = new OrderListener(this);
    
    // 加载初始订单数据
    this.loadInitialOrders();
  },

  onShow() {
    // 每次页面显示时启动监听
    if (!this.orderListener.isActive) {
      this.orderListener.start({
        status: this.data.selectedStatus
      });
    }
  },

  onHide() {
    // 页面进入后台时停止监听
    if (this.orderListener.isActive) {
      this.orderListener.stop();
    }
  },

  onUnload() {
    // 页面卸载时清理资源
    if (this.orderListener) {
      this.orderListener.stop();
    }
  },

  // 加载初始订单数据
  async loadInitialOrders() {
    try {
      const db = wx.cloud.database();
      const openid = wx.getStorageSync('openid');

      const query = { openid };
      if (this.data.selectedStatus && this.data.selectedStatus !== 'all') {
        query.status = this.data.selectedStatus;
      }

    ### 方法二：使用 WatcherMixin（推荐用于简单页面）

    **优点**: 自动管理监听生命周期，无需手动编写 onShow/onHide/onUnload

    ```javascript
    import WatcherMixin from '../../utils/watcher-mixin';

    Page({
      ...WatcherMixin,  // 混入生命周期自动管理

      data: {
        orders: [],
        selectedStatus: "all"
      },

      onLoad(options) {
        if (options?.status) {
          this.setData({ selectedStatus: options.status });
        }

        // 添加订单监听（自动在 onShow 时启动，onHide 时停止，onUnload 时销毁）
        this.addWatcher(() => {
          const db = wx.cloud.database();
          const openid = wx.getStorageSync('openid');
          const query = { openid };
      
          if (this.data.selectedStatus !== 'all') {
            query.status = this.data.selectedStatus;
          }

          return db.collection('orders')
            .where(query)
            .orderBy('createdAt', 'desc')
            .watch({
              onChange: (snapshot) => {
                const orders = snapshot.docs;
                this.setData({ orders });
              },
              onError: (err) => {
                console.error('订单监听失败:', err);
              }
            });
        });

        this.loadInitialOrders();
      },

      // onShow/onHide/onUnload 已由 WatcherMixin 自动处理

      async loadInitialOrders() {
        // ... 加载初始数据
      },

      switchStatus(e) {
        const status = e.currentTarget.dataset.status;
        this.setData({ selectedStatus: status });
        // WatcherMixin 会自动重新启动监听
      }
    });

## 第二步：消息页改造示例

### 修改 `pages/message/index.js`（会话列表）

```javascript
import { MessageListener } from '../../utils/messageListener';

Page({
  data: {
    sessions: [],
    messages: [],
    refreshing: false,
    // ... 其他数据
  },

  onLoad(options) {
    this.messageListener = new MessageListener(this);
    this.loadInitialSessions();
  },

  onShow() {
    // 启动会话监听（会话列表变化）
    this.messageListener.startSessionListener();
    
    // 加载初始会话数据
    if (this.data.sessions.length === 0) {
      this.loadInitialSessions();
    }
  },

  onUnload() {
    this.messageListener.stopAll();
  },

  async loadInitialSessions() {
    try {
      const db = wx.cloud.database();
      const openid = wx.getStorageSync('openid');

      const res = await db.collection('sessions')
        .where({ userId: openid })
        .orderBy('updatedAt', 'desc')
        .limit(50)
        .get();

      this.setData({
        sessions: res.data
      });

      console.log('[Message] 初始会话加载完成，共', res.data.length, '个');
    } catch (error) {
      console.error('[Message] 加载初始会话失败:', error);
    }
  }
});
```

### 修改 `pages/message/service/index.js`（聊天详情页）

```javascript
import { MessageListener } from '../../utils/messageListener';

Page({
  data: {
    sessionId: '',
    messages: [],
    messageListReady: false,
    openid: '',
    messageRole: '', // 'user' 或 'customer_service'
    // ... 其他数据
  },

  onLoad(options) {
    const sessionId = options.sessionId;
    this.setData({
      sessionId,
      openid: wx.getStorageSync('openid')
    });

    this.messageListener = new MessageListener(this);
    this.loadInitialMessages();
  },

  onShow() {
    // 启动消息监听
    this.messageListener.startMessageListener(this.data.sessionId);
    this.messageListener.startSessionListener();
  },

  onHide() {
    // 停止监听（保存连接状态）
    // 可选：完全停止监听
    // this.messageListener.stopAll();
  },

  onUnload() {
    // 卸载时完全停止监听
    this.messageListener.stopAll();
  },

  async loadInitialMessages() {
    try {
      const db = wx.cloud.database();
      
      const res = await db.collection('messages')
        .where({ sessionId: this.data.sessionId })
        .orderBy('createTime', 'desc')
        .limit(50)
        .get();

      const messages = res.data.reverse();
      
      this.setData({
        messages,
        messageListReady: true
      });

      // 刷新消息缓冲
      this.messageListener.flushMessageBuffer();

      // 滚动到底部
      setTimeout(() => {
        this.scrollToBottom();
      }, 100);

      console.log('[Chat] 初始消息加载完成，共', messages.length, '条');
    } catch (error) {
      console.error('[Chat] 加载初始消息失败:', error);
    }
  },

  scrollToBottom() {
    this.setData({
      scrollTop: 999999
    });
  },

  // 分组消息（如果已有此方法，保留不变）
  groupMessages(messages) {
    // 返回分组后的消息
    return messages;
  }
});

## 第三步：全局初始化（可选但推荐）

### 修改 `app.js`

```javascript
import { networkManager } from './miniprogram/utils/networkManager';
import { resourceManager } from './miniprogram/utils/resourceManager';

App({
  onLaunch() {
    this.globalData = {
      env: "cloud1-4gs2vu8c6544e586",
      openid: '',
      productsNeedRefresh: false,
      
      // 实时监听相关
      networkManager: networkManager,
      resourceManager: resourceManager,
      realtimeListeners: {}
    };

    // 初始化云开发
    if (!wx.cloud) {
      console.error("请使用 2.2.3 或以上的基础库以使用云能力");
    } else {
      wx.cloud.init({
        env: this.globalData.env,
        traceUser: true
      });
    }

    // 获取用户 openid
    this.getOpenid();

    // 监听网络状态变化
    networkManager.onChange((isOnline) => {
      console.log(`[App] 网络状态: ${isOnline ? '在线' : '离线'}`);
    });
  },

  getOpenid() {
    try {
      wx.cloud.callFunction({
        name: 'login',
        data: {},
        success: res => {
          console.log('获取openid成功:', res.result.openid);
          this.globalData.openid = res.result.openid;
          wx.setStorageSync('openid', res.result.openid);
        },
        fail: err => {
          console.error('获取openid失败:', err);
          this.globalData.openid = '';
        }
      });
    } catch (error) {
      console.error('调用云函数时发生错误:', error);
    }
  },

  onUnload() {
    // 应用卸载时清理所有资源
    if (this.globalData.resourceManager) {
      this.globalData.resourceManager.unloadAll();
    }
  }
});

### 缓存示例

```javascript
import { SmartCache } from '../../utils/optimizationUtils';

// 创建缓存实例（最多100项，5分钟过期）
const productCache = new SmartCache(100, 5 * 60 * 1000);

async function getProduct(productId) {
  // 先检查缓存
  const cached = productCache.get(`product_${productId}`);
  if (cached) {
    console.log('从缓存读取商品:', productId);
    return cached;
  }

  // 缓存未命中，从数据库获取
  const db = wx.cloud.database();
  const res = await db.collection('products').doc(productId).get();
  
  // 存入缓存
  productCache.set(`product_${productId}`, res.data);
  
  return res.data;
}
```

## 故障排查

### 问题 1：消息重复

**原因**: 本人消息既本地追加又通过 watch 接收

**解决方案**: 在消息监听回调中检查是否是本人消息

```javascript
const isOwnMessage = msg.openid === this.page.data.openid;
if (!isOwnMessage) {
  // 只处理对端消息
}
```

### 问题 2：内存持续增长

**原因**: 缓冲区未清理

**解决方案**: 监听器内置了自动清理，确保使用了框架提供的防抖/缓存工具

### 问题 3：网络恢复后数据不同步

**原因**: 监听断线后未重连

**解决方案**: 框架提供了自动重连机制，配置中设置 `autoReconnect: true`
