## API 速查表

### 核心监听管理器

```javascript
import { watch, unwatch, unwatchAll, getStatus, getAllStatus } from './utils/realtimeListener';

// 启动监听
watch(
  'orders_watch',                    // 监听标识
  { openid: 'xxx' },                 // 查询条件
  (changes, meta) => {               // 回调函数
    console.log(changes.added);      // 新增项
    console.log(changes.modified);   // 修改项
    console.log(changes.removed);    // 删除项
  }
);

// 停止单个监听
unwatch('orders_watch');

// 停止所有监听
unwatchAll();

// 获取单个监听状态
const status = getStatus('orders_watch');

// 获取所有监听状态
const allStatus = getAllStatus();
```

## 完整工作流

### 订单列表完整示例

```javascript
import { OrderListener } from '../../utils/orderListener';

Page({
  data: {
    orders: [],
    originalOrders: [],
    selectedStatus: 'all',
    loading: true
  },

  onLoad(options) {
    if (options?.status) {
      this.setData({ selectedStatus: options.status });
    }
    
    this.orderListener = new OrderListener(this);
    this.loadInitialOrders();
  },

  onShow() {
    if (!this.orderListener.isActive) {
      this.orderListener.start({
        status: this.data.selectedStatus
      });
    }
  },

  onHide() {
    if (this.orderListener.isActive) {
      this.orderListener.stop();
    }
  },

  onUnload() {
    this.orderListener?.stop();
  },

  async loadInitialOrders() {
    try {
      const db = wx.cloud.database();
      const openid = wx.getStorageSync('openid');
      
      const query = { openid };
      if (this.data.selectedStatus !== 'all') {
        query.status = this.data.selectedStatus;
      }

      const res = await db.collection('orders')
        .where(query)
        .orderBy('createdAt', 'desc')
        .limit(20)
        .get();

      this.setData({
        orders: res.data,
        originalOrders: res.data,
        loading: false
      });
    } catch (error) {
      console.error('加载失败:', error);
      wx.showToast({ title: '加载失败', icon: 'none' });
      this.setData({ loading: false });
    }
  },

  switchStatus(e) {
    const status = e.currentTarget.dataset.status;
    this.setData({ selectedStatus: status });
    
    if (this.orderListener.isActive) {
      this.orderListener.stop();
    }
    
    this.orderListener.start({ status });
  }
});
```

### 消息页完整示例

```javascript
import { MessageListener } from '../../utils/messageListener';

Page({
  data: {
    sessionId: '',
    messages: [],
    messageListReady: false,
    openid: ''
  },

  onLoad(options) {
    const sessionId = options.sessionId;
    this.setData({
      sessionId,
      openid: wx.getStorageSync('openid')
    });

    this.messageListener = new MessageListener(this);
    this.loadInitialMessages();
  },

  onShow() {
    this.messageListener.startMessageListener(this.data.sessionId);
    this.messageListener.startSessionListener();
  },

  onUnload() {
    this.messageListener.stopAll();
  },

  async loadInitialMessages() {
    try {
      const db = wx.cloud.database();
      const res = await db.collection('messages')
        .where({ sessionId: this.data.sessionId })
        .orderBy('createTime', 'desc')
        .limit(50)
        .get();

      const messages = res.data.reverse();
      
      this.setData({
        messages,
        messageListReady: true
      });

      this.messageListener.flushMessageBuffer();
      
      setTimeout(() => this.scrollToBottom(), 100);
    } catch (error) {
      console.error('加载失败:', error);
    }
  },

  scrollToBottom() {
    this.setData({ scrollTop: 999999 });
  },

  groupMessages(messages) {
    // 消息分组逻辑
    return messages;
  }
});
```
### 🆕 使用 WatcherMixin 的订单列表完整示例

```javascript
import WatcherMixin from '../../utils/watcherMixin';
import SmartCache from '../../utils/smartCache';

const orderCache = new SmartCache('orders_cache');

Page({
  ...WatcherMixin,  // 自动管理监听生命周期

  data: {
    orders: [],
    selectedStatus: 'all',
    loading: true
  },

  onLoad(options) {
    if (options?.status) {
      this.setData({ selectedStatus: options.status });
    }

    // 优先显示缓存数据
    this.loadFromCache();
    
    // 添加实时监听（自动在 onShow 启动，onHide 停止，onUnload 销毁）
    this.addWatcher(() => {
      const db = wx.cloud.database();
      const openid = wx.getStorageSync('openid');
      const query = { openid };
      
      if (this.data.selectedStatus !== 'all') {
        query.status = this.data.selectedStatus;
      }

      return db.collection('orders')
        .where(query)
        .orderBy('createdAt', 'desc')
        .watch({
          onChange: (snapshot) => {
            const orders = snapshot.docs;
            this.setData({ orders });
            
            // 更新缓存
            orderCache.set(orders);
          },
          onError: (err) => {
            console.error('订单监听失败:', err);
          }
        });
    });

    // 加载初始数据
    this.loadInitialOrders();
  },

  loadFromCache() {
    const cached = orderCache.get();
    if (cached) {
      this.setData({
        orders: cached,
        loading: false
      });
    }
  },

  async loadInitialOrders() {
    try {
      const db = wx.cloud.database();
      const openid = wx.getStorageSync('openid');
      const query = { openid };
      
      if (this.data.selectedStatus !== 'all') {
        query.status = this.data.selectedStatus;
      }

      const res = await db.collection('orders')
        .where(query)
        .orderBy('createdAt', 'desc')
        .limit(20)
        .get();

      this.setData({
        orders: res.data,
        loading: false
      });

      // 更新缓存
      orderCache.set(res.data);
    } catch (error) {
      console.error('加载失败:', error);
      this.setData({ loading: false });
    }
  },

  switchStatus(e) {
    const status = e.currentTarget.dataset.status;
    this.setData({ selectedStatus: status });
    // WatcherMixin 会自动重启监听
  }
});
```

## 监听生命周期

```
页面加载
  ↓
onLoad() → 初始化监听器 → 加载初始数据
  ↓
onShow() → 启动监听
  ↓
数据实时更新（watch 回调）
  ↓
onHide() → 暂停/停止监听
  ↓
onUnload() → 完全停止监听、清理资源
```

## 常见错误

| 错误 | 原因 | 解决方案 |
|------|------|---------|
| 监听已存在 | 重复启动监听 | 先调用 unwatch() 再重新 watch() |
| 消息重复 | 本人消息重复计算 | 检查 isOwnMessage 逻辑 |
| 内存泄漏 | 监听未正确卸载 | 确保 onUnload 中调用 stop() |
| 网络断连 | 网络波动 | 框架自动重连，注意日志 |
| 数据不同步 | 监听断线未重连 | 检查 onShow 中是否重启监听 |

## 最佳实践

✅ **DO**
- 在 onShow 中启动监听
- 在 onUnload 中停止监听
- 检查 openid 存在再启动监听
- 使用防抖处理频繁事件
- 监控资源和内存占用

❌ **DON'T**
- 重复启动相同的监听
- 忘记卸载监听导致资源泄漏
- 在 watch 回调中执行重型操作
- 同时启动过多监听（> 5 个）
- 监听全集合而不做条件过滤
